Future<Uint8List> buildSignedSosPacket({
required String touristTuid,
required double lat,
required double lng,
int hopCount = 0,
}) async {
final packetId = _generatePacketId();
final jwtSecret = await _secureStorage.read(key: 'jwt_token') ?? '';
 
// Build payload bytes BEFORE signing
final payload = ByteData(20);
payload.setFloat32(0, lat, Endian.big);
payload.setFloat32(4, lng, Endian.big);
payload.setUint8(8, hopCount);
final payloadBytes = Uint8List.sublistView(payload, 0, 9);
 
// Compute HMAC-SHA256, take first 8 bytes
final key = utf8.encode(jwtSecret);
final data = utf8.encode('$packetId:$touristTuid:') + payloadBytes;
final hmac = Hmac(sha256, key).convert(data);
final sig = Uint8List.fromList(hmac.bytes.sublist(0, 8));
 
// Assemble final packet
final packet = Uint8List(28);
final idBytes = utf8.encode(packetId.substring(0, 4));
packet.setRange(0, 4, idBytes);
final tuidBytes = utf8.encode(touristTuid.substring(0, 8));
packet.setRange(4, 12, tuidBytes);
packet.setRange(12, 21, payloadBytes);
packet.setRange(20, 28, sig);
return packet;
}

def verify_ble_sos_signature(
packet_id: str,
tourist_tuid: str,
payload_bytes: bytes,
received_sig: bytes,
jwt_secret: str,
) -> bool:
key = jwt_secret.encode()
data = f'{packet_id}:{tourist_tuid}:'.encode() + payload_bytes
expected = hmac.new(key, data, hashlib.sha256).digest()[:8]
return hmac.compare_digest(expected, received_sig)

class HeadingSmoother {
double _smoothedHeading = 0;
static const _alpha = 0.1; // low = smoother, high = more responsive
 
double update(double rawHeading) {
// Handle 359->0 wraparound
double delta = rawHeading - _smoothedHeading;
if (delta > 180) delta -= 360;
if (delta < -180) delta += 360;
_smoothedHeading = (_smoothedHeading + _alpha * delta) % 360;
return _smoothedHeading;
}
